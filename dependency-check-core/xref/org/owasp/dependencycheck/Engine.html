144     /**
145      * Loads the analyzers specified in the configuration file (or system
146      * properties).
147      */
148     private void loadAnalyzers() {
149         if (!analyzers.isEmpty()) {
150             return;
151         }
152         for (AnalysisPhase phase : AnalysisPhase.values()) {
153             analyzers.put(phase, new ArrayList<Analyzer>());
154         }
155 
156         final AnalyzerService service = new AnalyzerService(serviceClassLoader);
157         final List<Analyzer> iterator = service.getAnalyzers();
158         for (Analyzer a : iterator) {
159             analyzers.get(a.getAnalysisPhase()).add(a);
160             if (a instanceof FileTypeAnalyzer) {
161                 this.fileTypeAnalyzers.add((FileTypeAnalyzer) a);
162             }
163         }
164     }

406                     final Dependency d = scanFile(f, projectReference);
407                     deps.add(d);
408                 }
409             }
410         }
411         return deps;

425     /**
426      * Scans a specified file. If a dependency is identified it is added to the
427      * dependency collection.
428      *
429      * @param file The file to scan
430      * @param projectReference the name of the project or scope in which the
431      * dependency was identified
432      * @return the scanned dependency
433      * @since v1.4.4
434      */
435     protected Dependency scanFile(File file, String projectReference) {
436         Dependency dependency = null;
437         if (file.isFile()) {
438             if (accept(file)) {
439                 dependency = new Dependency(file);
440                 if (projectReference != null) {
441                     dependency.addProjectReference(projectReference);
442                 }
443                 final String sha1 = dependency.getSha1sum();
444                 boolean found = false;
445                 synchronized (dependencies) {
446                     if (sha1 != null) {
447                         for (Dependency existing : dependencies) {
448                             if (sha1.equals(existing.getSha1sum())) {
449                                 found = true;
450                                 if (projectReference != null) {
451                                     existing.addProjectReference(projectReference);
452                                 }
453                                 if (existing.getActualFilePath() != null && dependency.getActualFilePath() != null
454                                         && !existing.getActualFilePath().equals(dependency.getActualFilePath())) {
455                                     existing.addRelatedDependency(dependency);
456                                 } else {
457                                     dependency = existing;
458                                 }
459                                 break;
460                             }
461                         }
462                     }
463                     if (!found) {
464                         dependencies.add(dependency);
465                     }
466                 }
467             } else {
468                 LOGGER.debug("Path passed to scanFile(File) is not a file: {}. Skipping the file.", file);
469             }
470         }
471         return dependency;
472     }

474     /**
475      * Runs the analyzers against all of the dependencies. Since the mutable
476      * dependencies list is exposed via {@link #getDependencies()}, this method
477      * iterates over a copy of the dependencies list. Thus, the potential for
478      * {@link java.util.ConcurrentModificationException}s is avoided, and
479      * analyzers may safely add or remove entries from the dependencies list.
480      * <p>
481      * Every effort is made to complete analysis on the dependencies. In some
482      * cases an exception will occur with part of the analysis being performed
483      * which may not affect the entire analysis. If an exception occurs it will
484      * be included in the thrown exception collection.
485      *
486      * @throws ExceptionCollection a collections of any exceptions that occurred
487      * during analysis
488      */
489     public void analyzeDependencies() throws ExceptionCollection {
490         final List<Throwable> exceptions = Collections.synchronizedList(new ArrayList<Throwable>());
491 
492         initializeAndUpdateDatabase(exceptions);
493 
494         //need to ensure that data exists
495         try {
496             ensureDataExists();
497         } catch (NoDataException ex) {
498             throwFatalExceptionCollection("Unable to continue dependency-check analysis.", ex, exceptions);
499         }
500 
501         LOGGER.debug("\n----------------------------------------------------\nBEGIN ANALYSIS\n----------------------------------------------------");
502         LOGGER.info("Analysis Started");
503         final long analysisStart = System.currentTimeMillis();
504 
505         // analysis phases
506         for (AnalysisPhase phase : AnalysisPhase.values()) {
507             final List<Analyzer> analyzerList = analyzers.get(phase);
508 
509             for (final Analyzer analyzer : analyzerList) {
510                 final long analyzerStart = System.currentTimeMillis();
511                 try {
512                     initializeAnalyzer(analyzer);
513                 } catch (InitializationException ex) {
514                     exceptions.add(ex);
515                     continue;
516                 }
517 
518                 if (analyzer.isEnabled()) {
519                     executeAnalysisTasks(analyzer, exceptions);
520 
521                     final long analyzerDurationMillis = System.currentTimeMillis() - analyzerStart;
522                     final long analyzerDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(analyzerDurationMillis);
523                     LOGGER.info("Finished {} ({} seconds)", analyzer.getName(), analyzerDurationSeconds);
524                 } else {
525                     LOGGER.debug("Skipping {} (not enabled)", analyzer.getName());
526                 }
527             }
528         }
529         for (AnalysisPhase phase : AnalysisPhase.values()) {
530             final List<Analyzer> analyzerList = analyzers.get(phase);
531 
532             for (Analyzer a : analyzerList) {
533                 closeAnalyzer(a);
534             }
535         }
536 
537         LOGGER.debug("\n----------------------------------------------------\nEND ANALYSIS\n----------------------------------------------------");
538         final long analysisDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(System.currentTimeMillis() - analysisStart);
539         LOGGER.info("Analysis Complete ({} seconds)", analysisDurationSeconds);
540         if (exceptions.size() > 0) {
541             throw new ExceptionCollection("One or more exceptions occurred during dependency-check analysis", exceptions);
542         }
543     }

545     /**
546      * Performs any necessary updates and initializes the database.
547      *
548      * @param exceptions a collection to store non-fatal exceptions
549      * @throws ExceptionCollection thrown if fatal exceptions occur
550      */
551     private void initializeAndUpdateDatabase(final List<Throwable> exceptions) throws ExceptionCollection {
552         boolean autoUpdate = true;
553         try {
554             autoUpdate = Settings.getBoolean(Settings.KEYS.AUTO_UPDATE);
555         } catch (InvalidSettingException ex) {
556             LOGGER.debug("Invalid setting for auto-update; using true.");
557             exceptions.add(ex);
558         }
559         if (autoUpdate) {
560             try {
561                 database = CveDB.getInstance();
562                 doUpdates();
563             } catch (UpdateException ex) {
564                 exceptions.add(ex);
565                 LOGGER.warn("Unable to update Cached Web DataSource, using local "
566                         + "data instead. Results may not include recent vulnerabilities.");
567                 LOGGER.debug("Update Error", ex);
568             } catch (DatabaseException ex) {
569                 throw new ExceptionCollection("Unable to connect to the database", ex);
570             }
571         } else {
572             try {
573                 if (ConnectionFactory.isH2Connection() && !ConnectionFactory.h2DataFileExists()) {
574                     throw new ExceptionCollection(new NoDataException("Autoupdate is disabled and the database does not exist"), true);
575                 } else {
576                     database = CveDB.getInstance();
577                 }
578             } catch (IOException ex) {
579                 throw new ExceptionCollection(new DatabaseException("Autoupdate is disabled and unable to connect to the database"), true);
580             } catch (DatabaseException ex) {
581                 throwFatalExceptionCollection("Unable to connect to the dependency-check database.", ex, exceptions);
582             }
583         }
584     }

586     /**
587      * Executes executes the analyzer using multiple threads.
588      *
589      * @param exceptions a collection of exceptions that occurred during
590      * analysis
591      * @param analyzer the analyzer to execute
592      * @throws ExceptionCollection thrown if exceptions occurred during analysis
593      */
594     protected void executeAnalysisTasks(Analyzer analyzer, List<Throwable> exceptions) throws ExceptionCollection {
595         LOGGER.debug("Starting {}", analyzer.getName());
596         final List<AnalysisTask> analysisTasks = getAnalysisTasks(analyzer, exceptions);
597         final ExecutorService executorService = getExecutorService(analyzer);
598 
599         try {
600             final List<Future<Void>> results = executorService.invokeAll(analysisTasks, 10, TimeUnit.MINUTES);
601 
602             // ensure there was no exception during execution
603             for (Future<Void> result : results) {
604                 try {
605                     result.get();
606                 } catch (ExecutionException e) {
607                     throwFatalExceptionCollection("Analysis task failed with a fatal exception.", e, exceptions);
608                 } catch (CancellationException e) {
609                     throwFatalExceptionCollection("Analysis task timed out.", e, exceptions);
610                 }
611             }
612         } catch (InterruptedException e) {
613             Thread.currentThread().interrupt();
614             throwFatalExceptionCollection("Analysis has been interrupted.", e, exceptions);
615         } finally {
616             executorService.shutdown();
617         }
618     }

620     /**
621      * Returns the analysis tasks for the dependencies.
622      *
623      * @param analyzer the analyzer to create tasks for
624      * @param exceptions the collection of exceptions to collect
625      * @return a collection of analysis tasks
626      */
627     protected List<AnalysisTask> getAnalysisTasks(Analyzer analyzer, List<Throwable> exceptions) {
628         final List<AnalysisTask> result = new ArrayList<>();
629         synchronized (dependencies) {
630             for (final Dependency dependency : dependencies) {
631                 final AnalysisTask task = new AnalysisTask(analyzer, dependency, this, exceptions, Settings.getInstance());
632                 result.add(task);
633             }
634         }
635         return result;
636     }

638     /**
639      * Returns the executor service for a given analyzer.
640      *
641      * @param analyzer the analyzer to obtain an executor
642      * @return the executor service
643      */
644     protected ExecutorService getExecutorService(Analyzer analyzer) {
645         if (analyzer.supportsParallelProcessing()) {
646             final int maximumNumberOfThreads = Runtime.getRuntime().availableProcessors();
647             LOGGER.debug("Parallel processing with up to {} threads: {}.", maximumNumberOfThreads, analyzer.getName());
648             return Executors.newFixedThreadPool(maximumNumberOfThreads);
649         } else {
650             LOGGER.debug("Parallel processing is not supported: {}.", analyzer.getName());
651             return Executors.newSingleThreadExecutor();
652         }
653     }

655     /**
656      * Initializes the given analyzer.
657      *
658      * @param analyzer the analyzer to initialize
659      * @throws InitializationException thrown when there is a problem
660      * initializing the analyzer
661      */
662     protected void initializeAnalyzer(Analyzer analyzer) throws InitializationException {
663         try {
664             LOGGER.debug("Initializing {}", analyzer.getName());
665             analyzer.initialize();
666         } catch (InitializationException ex) {
667             LOGGER.error("Exception occurred initializing {}.", analyzer.getName());
668             LOGGER.debug("", ex);
669             try {
670                 analyzer.close();
671             } catch (Throwable ex1) {
672                 LOGGER.trace("", ex1);
673             }
674             throw ex;
675         } catch (Throwable ex) {
676             LOGGER.error("Unexpected exception occurred initializing {}.", analyzer.getName());
677             LOGGER.debug("", ex);
678             try {
679                 analyzer.close();
680             } catch (Throwable ex1) {
681                 LOGGER.trace("", ex1);
682             }
683             throw new InitializationException("Unexpected Exception", ex);
684         }
685     }

701     /**
702      * Cycles through the cached web data sources and calls update on all of
703      * them.
704      *
705      * @throws UpdateException thrown if the operation fails
706      */
707     public void doUpdates() throws UpdateException {
708         LOGGER.info("Checking for updates");
709         final long updateStart = System.currentTimeMillis();
710         final UpdateService service = new UpdateService(serviceClassLoader);
711         final Iterator<CachedWebDataSource> iterator = service.getDataSources();
712         while (iterator.hasNext()) {
713             final CachedWebDataSource source = iterator.next();
714             source.update();
715         }
716         LOGGER.info("Check for updates complete ({} ms)", System.currentTimeMillis() - updateStart);
717     }

719     /**
720      * Returns a full list of all of the analyzers. This is useful for reporting
721      * which analyzers where used.
722      *
723      * @return a list of Analyzers
724      */
725     public List<Analyzer> getAnalyzers() {
726         final List<Analyzer> ret = new ArrayList<>();
727         for (AnalysisPhase phase : AnalysisPhase.values()) {
728             final List<Analyzer> analyzerList = analyzers.get(phase);
729             ret.addAll(analyzerList);
730         }
731         return ret;
732     }

734     /**
735      * Checks all analyzers to see if an extension is supported.
736      *
737      * @param file a file extension
738      * @return true or false depending on whether or not the file extension is
739      * supported
740      */
741     @Override
742     public boolean accept(File file) {
743         if (file == null) {
744             return false;
745         }
746         boolean scan = false;
747         for (FileTypeAnalyzer a : this.fileTypeAnalyzers) {
748             /* note, we can't break early on this loop as the analyzers need to know if
749              they have files to work on prior to initialization */
750             scan |= a.accept(file);
751         }
752         return scan;
753     }
